  // 삭제 함수
  const deleteRequest = async (id: string) => {
    if (!confirm('이 견적 요청을 삭제하시겠습니까?')) {
      return;
    }

    try {
      // 🔥 Supabase requests 테이블에서 직접 삭제
      const { projectId, publicAnonKey } = await import('/utils/supabase/info');
      const supabaseUrl = `https://${projectId}.supabase.co`;
      const supabase = createClient(supabaseUrl, publicAnonKey);
      
      console.log('🗑️ Supabase requests 테이블에서 삭제 시도:', id);
      
      const { error: deleteError } = await supabase
        .from('requests')
        .delete()
        .eq('id', id);
      
      if (deleteError) {
        console.error('❌ Supabase 삭제 실패:', deleteError);
        throw new Error(deleteError.message);
      }
      
      console.log('✅ 견적 요청 삭제 완료:', id);
      alert('견적 요청이 삭제되었습니다.');
      
      // 목록 새로고침
      fetchRequests();
      
      // 상세 모달이 열려있으면 닫기
      if (showDetailModal && selectedRequest?.id === id) {
        setShowDetailModal(false);
        setSelectedRequest(null);
      }
    } catch (err) {
      console.error('❌ 삭제 오류:', err);
      alert(`삭제 중 오류가 발생했습니다.\n\n상세 오류: ${err}`);
    }
  };
